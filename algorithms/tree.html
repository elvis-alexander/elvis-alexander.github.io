public Node mergeTrees(Node n1, Node n2) {

        if(n1 == null && n2 == null)
            return null;

        int val = (n1 == null) ? 0 : n1.val;
        val += (n2 == null) ? 0 : n2.val;

        Node n = (n1 == null) ? n2 : n1;
        n.val = val;

        n.left = mergeTrees(n1 == null ? null : n1.left, n2 == null ? null : n2.left);
        n.right = mergeTrees(n1 == null ? null : n1.right, n2 == null ? null : n2.right);
        return n;
}




public TreeNode trimBST(TreeNode root, int L, int R) {
        if(root == null)
            return null;
        boolean inRange = root.val >= L && root.val <= R;
        if(! inRange)
            return trimBST(root.val < L ? root.right : root.left, L, R);
        root.left = trimBST(root.left, L, R);
        root.right = trimBST(root.right, L, R);
        return root;
}



public int maxDepth(TreeNode root) {
    return root == null ? 0 : (1 + Math.max(maxDepth(root.left), maxDepth(root.right)));
}



public TreeNode invertTree(TreeNode root) {
    if(root == null)
        return null;
    TreeNode original_left = root.left;
    root.left = invertTree(root.right);
    root.right = invertTree(original_left);
    return root;
}



public String tree2str(TreeNode t) {
    // base cases
    if(t == null)
        return "";
    boolean isLeaf = t.left == null && t.right == null;
    if(isLeaf) return ""+t.val;

    // recursive case
    String init = String.format("%s(%s)", ""+t.val, tree2str(t.left));
    if(t.right != null)
        init += String.format("(%s)", tree2str(t.right));
    return init;
}
