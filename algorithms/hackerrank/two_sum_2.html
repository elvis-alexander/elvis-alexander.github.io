int[] twoSum(int[] arr, int target) {
	int[] out = new int[2];
	
	for(int i = 0; i < arr.length; ++i) {
		for(int j = i+1; j < arr.length; ++j) {
			if(arr[i] + arr[j] == target) {
				return new int[]{i+1, j+1};
			}
		}					
	}    	
	return out;     
}
